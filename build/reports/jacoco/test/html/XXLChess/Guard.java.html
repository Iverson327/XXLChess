public class Guard extends Piece{
    public static final String bImage = "src/main/resources/XXLChess/b-knight-king.png";
    public static final String wImage = "src/main/resources/XXLChess/w-knight-king.png";
    
    public int[][] validMove(int row, int col){
        int[][] moves = new int[14][14];
        if(row + 1 < 14){
            moves[row + 1][col] = 1;
            if(col + 1 < 14){
                moves[row][col + 1] = 1;
                moves[row + 1][col + 1] = 1;
            }
            if(col - 1 >= 0){
                moves[row][col - 1] = 1;
                moves[row + 1][col - 1] = 1;
            }
        }
        if(row - 1 >= 0){
            moves[row - 1][col] = 1;
            if(col + 1 < 14){
                moves[row - 1][col + 1] = 1;
            }
            if(col - 1 >= 0){
                moves[row - 1][col - 1] = 1;
            }
        }
        if(col + 1 < 14){
            moves[row][col + 1] = 1;
        }
        if(col - 1 >= 0){
            moves[row][col - 1] = 1;
        }
        if(row + 2 < 14){
            if(col + 1 < 14){
                moves[row + 2][col + 1] = 1;
            }
            if(col - 1 >= 0){
                moves[row + 2][col - 1] = 1;
            }
        }
        if(row - 2 >= 0){
            if(col + 1 < 14){
                moves[row - 2][col + 1] = 1;
            }
            if(col - 1 >= 0){
                moves[row - 2][col - 1] = 1;
            }
        }
        if(col + 2 < 14){
            if(row + 1 < 14){
                moves[row + 1][col + 2] = 1;
            }
            if(row - 1 >= 0){
                moves[row - 1][col + 2] = 1;
            }
        }
        if(col - 2 >= 0){
            if(row + 1 < 14){
                moves[row + 1][col - 2] = 1;
            }
            if(row - 1 >= 0){
                moves[row - 1][col - 2] = 1;
            }
        }

        moves[row][col] = 2;
        return moves;
    }
}
