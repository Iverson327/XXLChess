public class Amazon extends Piece{
    public static final String bImage = "src/main/resources/XXLChess/b-amazon.png";
    public static final String wImage = "src/main/resources/XXLChess/w-amazon.png";

    public int[][] validMove(int row, int col){
        int[][] moves = new int[14][14];
        for(int i = 0; i < 14; i++){
            for(int j = 0; j < 14; j++){
                if(i == row){
                    moves[i][j] = 1;
                }
                if(j == col){
                    moves[i][j] = 1;
                }
                if(col - j == row - i || j - col == row - i){
                    moves[i][j] = 1;
                }  
            }
        }
        if(row + 2 < 14){
            if(col + 1 < 14){
                moves[row + 2][col + 1] = 1;
            }
            if(col - 1 >= 0){
                moves[row + 2][col - 1] = 1;
            }
        }
        if(row - 2 >= 0){
            if(col + 1 < 14){
                moves[row - 2][col + 1] = 1;
            }
            if(col - 1 >= 0){
                moves[row - 2][col - 1] = 1;
            }
        }
        if(col + 2 < 14){
            if(row + 1 < 14){
                moves[row + 1][col + 2] = 1;
            }
            if(row - 1 >= 0){
                moves[row - 1][col + 2] = 1;
            }
        }
        if(col - 2 >= 0){
            if(row + 1 < 14){
                moves[row + 1][col - 2] = 1;
            }
            if(row - 1 >= 0){
                moves[row - 1][col - 2] = 1;
            }
        }

        moves[row][col] = 2;
        return moves;
    }
}
